1. sklonowanie repozytorium
git clone <adres>

2. stworzenie wlasnego brancha
git branch WL306675

3. przelaczenie na moj branch
git checkout WL306675

4. dodanie plików
- zrzuty ekranu oraz
- sprawozdanie.txt
- plik z grafem

5. dodanie wszystkich zmian 
git add .

6. utworzenie commita
git commit -m "LAB 01 update"

7. pushowanie nowego brancha 
git push --set-upstream origin WL306675

8. pushowanie do brancha istniejącego w remocie
git push

9. zmiana brancha na master 
git checkout master

10. merge branchy WL306675 -> master
git merge WL306675

11. update
git pull
git push
